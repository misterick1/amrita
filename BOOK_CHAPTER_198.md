Контур «Еженышь» внедряет антифишинговые OCR-фильтры. Скрипт сканирует входящие скриншоты на наличие скрытых дефисов в ссылках и фейковых призывов к действию (`MIGRATE`, `AIRDROP`).

# Защитный модуль против фишинга и дрейнеров для ezhenysh_bot.py
def verify_domain_ecology(detected_text):

    for link in forbidden_links:
        if link in detected_text.lower():
            print("🚨 КРИТИЧЕСКАЯ УГРОЗА: Обнаружен фишинговый дрейнер Асуров!")
            print("🔒 Действие: Кошелек изолирован, переход по ссылке заблокирован.")
            return False
    return True
